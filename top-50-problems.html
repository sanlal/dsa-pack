01 Two Sum · LC 1 The most asked question in interviewing history. easy4 min ▾
Google Amazon Meta Microsoft Apple Uber

Problem. Given an array nums and an integer target, return the indices of the two numbers that add up to target.

Approach. Brute force tries every pair (O(n²)). Instead, walk once and store each seen number in a hash map; for each new x, check if target - x is already there.

Key insight Trade O(n) space for O(n) time. The hash map answers "have I seen the complement?" in O(1).

Better Sort + two pointers. Quick to think of, but you lose the original indices — need a pair-tracking workaround. Time O(n log n)Space O(n)
Optimal Hash map of value → index. For each x, ask “have I seen target - x?” Time O(n)Space O(n) 
vector<int> twoSum(vector<int>& nums, int target) {
    unordered_map<int,int> seen;
    for (int i = 0; i < (int)nums.size(); ++i) {
        int need = target - nums[i];
        if (seen.count(need)) return {seen[need], i};
        seen[nums[i]] = i;
    }
    return {};
}
int[] twoSum(int[] nums, int target) {
    Map<Integer, Integer> seen = new HashMap<>();
    for (int i = 0; i < nums.length; i++) {
        int need = target - nums[i];
        if (seen.containsKey(need)) return new int[]{seen.get(need), i};
        seen.put(nums[i], i);
    }
    return new int[0];
}
def two_sum(nums, target):
    seen = {}
    for i, x in enumerate(nums):
        need = target - x
        if need in seen: return [seen[need], i]
        seen[x] = i

Problem. Return true if string t is an anagram of s.

Approach. Count letters in both. If counts match, they're anagrams. For lowercase English only, a fixed 26-int array beats a HashMap on speed and memory.

Key insight Use one array; ++ for s, -- for t. If any cell ends ≠ 0, not anagram.

bool isAnagram(string s, string t) {
    if (s.size() != t.size()) return false;
    int cnt[26] = {0};
    for (int i = 0; i < (int)s.size(); ++i) {
        cnt[s[i] - 'a']++;
        cnt[t[i] - 'a']--;
    }
    for (int c : cnt) if (c) return false;
    return true;
}
boolean isAnagram(String s, String t) {

        cnt[t.charAt(i) - 'a']--;
    }
    for (int c : cnt) if (c != 0) return false;
    return true;
}

vector<vector<string>> groupAnagrams(vector<string>& strs) {
    unordered_map<string, vector<string>> groups;
    for (auto& s : strs) {
        string key(26, '0');
        for (char c : s) key[c - 'a']++;
        groups[key].push_back(s);
    }
    vector<vector<string>> out;
    for (auto& [_, v] : groups) out.push_back(move(v));
    return out;
}
List<List<String>> groupAnagrams(String[] strs) {
    Map<String, List<String>> groups = new HashMap<>();

        int[] cnt = new int[26];
        for (char c : s.toCharArray()) cnt[c - 'a']++;
        String key = Arrays.toString(cnt);
        groups.computeIfAbsent(key, k -> new ArrayList<>()).add(s);
    }
    return new ArrayList<>(groups.values());
}

05 Top K Frequent Elements · LC 347 Count frequencies, then pull the loudest K. med6 min ▾
Amazon Meta Google Microsoft

Problem. Return the K most frequent elements.

Approach. Two great options: (a) Min-heap of size K — O(n log K). (b) Bucket sort by frequency in an array of length n+1 — O(n). Bucket sort wins when K is unknown / arbitrary.

Key insight Frequency is bounded by n, so an array of size n+1 indexed by count gives a free "sort by frequency" in linear time.

Brute Count frequencies, then sort the (value, count) pairs by count and take the first k. Time O(n log n)Space O(n)
Better Count frequencies and push them into a size-k min-heap; pop when it grows past k. Time O(n log k)Space O(n + k)
Optimal Bucket sort by frequency: buckets[freq] is the list of values that appear freq times. Walk buckets high→low. Time O(n)Space O(n) 
vector<int> topKFrequent(vector<int>& nums, int k) {
    unordered_map<int,int> freq;
    for (int x : nums) freq[x]++;
    int n = nums.size();
    vector<vector<int>> buckets(n + 1);
    for (auto& [val, f] : freq) buckets[f].push_back(val);
    vector<int> out;
    for (int f = n; f >= 0 && (int)out.size() < k; --f)
        for (int v : buckets[f]) {
            out.push_back(v);
            if ((int)out.size() == k) return out;
        }
    return out;
}
int[] topKFrequent(int[] nums, int k) {
    Map<Integer, Integer> freq = new HashMap<>();
    for (int x : nums) freq.merge(x, 1, Integer::sum);
    List<Integer>[] buckets = new List[nums.length + 1];
    for (var e : freq.entrySet()) {
        int f = e.getValue();
        if (buckets[f] == null) buckets[f] = new ArrayList<>();
        buckets[f].add(e.getKey());
    }
    int[] out = new int[k]; int idx = 0;
    for (int f = buckets.length - 1; f >= 0 && idx < k; f--)
        if (buckets[f] != null)
            for (int v : buckets[f]) {
                out[idx++] = v;
                if (idx == k) return out;
            }
    return out;
}
from collections import Counter
def top_k_frequent(nums, k):
    freq = Counter(nums)
    buckets = [[] for _ in range(len(nums) + 1)]

    out = []
    for f in range(len(buckets) - 1, -1, -1):
        for v in buckets[f]:
            out.append(v)
            if len(out) == k: return out

06 Product of Array Except Self · LC 238 No division allowed. Two passes — prefix and suffix. med6 min ▾
Amazon Meta Apple Microsoft

Problem. Return an array where out[i] = product of all elements except nums[i]. No division. O(n) time.

Approach. Two sweeps. First pass left-to-right: out[i] = product of everything to the left of i. Second pass right-to-left: multiply each by the running product of everything to the right.

Key insight Use the output array as the prefix-product buffer; a single running variable carries the suffix product. O(1) extra space (besides output).

Optimal Two sweeps: left-prefix product into ans, then a right-to-left pass multiplying by a running suffix product. Time O(n)Space O(1) output excl. 
vector<int> productExceptSelf(vector<int>& nums) {
    int n = nums.size();
    vector<int> out(n, 1);
    for (int i = 1; i < n; ++i) out[i] = out[i - 1] * nums[i - 1];
    int right = 1;
    for (int i = n - 1; i >= 0; --i) {
        out[i] *= right;
        right *= nums[i];
    }
    return out;
}

Problem. Design an algorithm to encode a list of strings into a single string. The encoded string can then be decoded back to the original list.

Approach. Prefix each string with its length followed by a delimiter (#). On decode, read the length, jump that many chars, repeat. Works even if the strings themselves contain #.

Key insight Don't try clever escaping. Length-prefix encoding is unambiguous and trivial to decode.

string encode(vector<string>& strs) {
    string out;
    for (auto& s : strs) out += to_string(s.size()) + "#" + s;
    return out;
}
vector<string> decode(string s) {
    vector<string> out;
    int i = 0;
    while (i < (int)s.size()) {
        int j = s.find('#', i);
        int len = stoi(s.substr(i, j - i));
        out.push_back(s.substr(j + 1, len));
        i = j + 1 + len;
    }
    return out;
}
String encode(List<String> strs) {
    StringBuilder sb = new StringBuilder();
    for (String s : strs) sb.append(s.length()).append('#').append(s);
    return sb.toString();
}
List<String> decode(String s) {
    List<String> out = new ArrayList<>();
    int i = 0;
    while (i < s.length()) {
        int j = s.indexOf('#', i);
        int len = Integer.parseInt(s.substring(i, j));
        out.add(s.substring(j + 1, j + 1 + len));
        i = j + 1 + len;
    }
    return out;
}
def encode(strs):
    return "".join(f"{len(s)}#{s}" for s in strs)

def decode(s):
    out, i = [], 0
    while i < len(s):
        j = s.find('#', i)
        ln = int(s[i:j])
        out.append(s[j + 1 : j + 1 + ln])

    return out

08 Longest Consecutive Sequence · LC 128 Find the longest run of consecutive integers - in O(n). med6 min ▾
Amazon Google Meta Microsoft

Problem. Given an unsorted array, find the length of the longest sequence of consecutive integers (e.g. [100, 4, 200, 1, 3, 2] → [1, 2, 3, 4] = 4).

Approach. Put all numbers in a HashSet. Only start counting a streak at a number x where x - 1 is absent (i.e., x is the bottom of its run). From each starter, walk up while the next is in the set.

Key insight The "start only at run-bottoms" trick keeps the inner walk amortised O(1) per element, giving overall O(n).

int longestConsecutive(vector<int>& nums) {
    unordered_set<int> s(nums.begin(), nums.end());
    int best = 0;
    for (int x : s) {
        if (s.count(x - 1)) continue;       // not a run-bottom
        int cur = x, len = 1;
        while (s.count(cur + 1)) { cur++; len++; }
        best = max(best, len);
    }
    return best;
}
int longestConsecutive(int[] nums) {

    for (int x : s) {
        if (s.contains(x - 1)) continue;

    }
    return best;
}
def longest_consecutive(nums):
    s = set(nums)
    best = 0
    for x in s:
        if x - 1 in s: continue
        cur, ln = x, 1
        while cur + 1 in s:
            cur += 1; ln += 1
        best = max(best, ln)
    return best

bool isPalindrome(string s) {
    int l = 0, r = s.size() - 1;
    while (l < r) {
        while (l < r && !isalnum(s[l])) l++;
        while (l < r && !isalnum(s[r])) r--;
        if (tolower(s[l]) != tolower(s[r])) return false;
        l++; r--;
    }
    return true;
}

}

vector<vector<int>> threeSum(vector<int>& nums) {
    sort(nums.begin(), nums.end());
    vector<vector<int>> out;
    int n = nums.size();
    for (int i = 0; i < n - 2; ++i) {
        if (i > 0 && nums[i] == nums[i - 1]) continue;
        int l = i + 1, r = n - 1;
        while (l < r) {
            int s = nums[i] + nums[l] + nums[r];
            if (s == 0) {
                out.push_back({nums[i], nums[l], nums[r]});
                while (l < r && nums[l] == nums[l + 1]) l++;
                while (l < r && nums[r] == nums[r - 1]) r--;
                l++; r--;
            } else if (s < 0) l++; else r--;
        }
    }
    return out;
}

int lengthOfLIS(int[] a) {

    for (int x : a) {
        int lo = 0, hi = tails.size();
        while (lo < hi) {
            int m = (lo + hi) / 2;
            if (tails.get(m) < x) lo = m + 1; else hi = m;
        }
        if (lo == tails.size()) tails.add(x); else tails.set(lo, x);
    }
    return tails.size();
}

int longestCommonSubsequence(string a, string b) {
    int n = a.size(), m = b.size();
    vector<vector<int>> dp(n + 1, vector<int>(m + 1, 0));
    for (int i = 1; i <= n; ++i)
        for (int j = 1; j <= m; ++j)
            dp[i][j] = a[i - 1] == b[j - 1]

int minDistance(string a, string b) {
    int n = a.size(), m = b.size();
    vector<vector<int>> dp(n + 1, vector<int>(m + 1, 0));
    for (int i = 0; i <= n; ++i) dp[i][0] = i;
    for (int j = 0; j <= m; ++j) dp[0][j] = j;
    for (int i = 1; i <= n; ++i)
        for (int j = 1; j <= m; ++j) {

bool canPartition(vector<int>& nums) {
    int total = accumulate(nums.begin(), nums.end(), 0);
    if (total % 2) return false;
    int target = total / 2;
    vector<bool> dp(target + 1, false);
    dp[0] = true;
    for (int x : nums)
        for (int c = target; c >= x; --c)
            dp[c] = dp[c] || dp[c - x];
    return dp[target];
}

def can_partition(nums):
    total = sum(nums)
    if total % 2: return False
    target = total // 2
    dp = [False] * (target + 1); dp[0] = True
    for x in nums:
        for c in range(target, x - 1, -1):
            dp[c] = dp[c] or dp[c - x]
    return dp[target]
